   Node reverseList(Node head)
   {
        // add code here
        Node prev, nextNode;
        prev=null;
        nextNode=null;
        while(head!=null)
        {
            nextNode=head.next;
            head.next=prev;
            prev=head;
            head=nextNode;
        }
        head=prev;
